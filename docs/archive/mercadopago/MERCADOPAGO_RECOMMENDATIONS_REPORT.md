### HIGH 1: Implement Asynchronous Webhook Processing

**Severity**: HIGH (Performance Risk)
**Impact**: Webhook timeouts under load, duplicate processing on retries
**Effort**: 2-3 hours
**Status**: ❌ Not Implemented

**Current Implementation**:
Webhooks process synchronously in the HTTP request handler (timeout risk).

File: `/Users/max/Documents/tifossi-expo/tifossi/backend/strapi/src/webhooks/mercadopago.ts` (lines 58-135)

```typescript
// CURRENT CODE (SYNCHRONOUS - RISKY)
async handleWebhook(ctx: Context): Promise<void> {
  try {
    // Signature verification (fast)
    if (!mpService.verifyWebhookSignature(...)) {
      ctx.unauthorized('Invalid signature');
      return;
    }

    // Payment processing (SLOW - can timeout)
    await this.handlePaymentNotification(webhookData.data, requestId);

    // Return 200 OK (only after processing completes)
    ctx.status = 200;
    ctx.body = { success: true };
  } catch (error) {
    // Error returns 200 to prevent retries (wrong approach)
    ctx.status = 200;
    ctx.body = { success: false, error: error.message };
  }
}
```

**Problem**:
1. If `handlePaymentNotification()` takes >30 seconds, request times out
2. MercadoPago retries failed webhooks, causing duplicates
3. Error handling returns 200 for unrecoverable errors (prevents retries)

**Recommended Fix**:

Implement background job queue using Bull/BullMQ:

```typescript
// RECOMMENDED IMPLEMENTATION (ASYNCHRONOUS)
import Queue from 'bull';

// Create webhook queue (in bootstrap)
const webhookQueue = new Queue('mercadopago-webhooks', {
  redis: process.env.REDIS_URL,
});

// Process jobs in background worker
webhookQueue.process(async (job) => {
  const { webhookData, requestId } = job.data;

  try {
    await this.handlePaymentNotification(webhookData.data, requestId);
    return { success: true };
  } catch (error) {
    // Job will retry with exponential backoff
    throw error;
  }
});

// Webhook handler (immediate response)
async handleWebhook(ctx: Context): Promise<void> {
  try {
    const webhookData = ctx.request.body;
    const signature = ctx.request.headers['x-signature'];
    const requestId = ctx.request.headers['x-request-id'];
    const dataId = webhookData?.data?.id?.toString();

    // Validate signature (fast)
    if (!mpService.verifyWebhookSignature(signature, requestId, dataId)) {
      ctx.unauthorized('Invalid signature');
      return;
    }

    // Check for duplicates (fast database query)
    if (await this.isDuplicateWebhook(requestId, webhookData)) {
      ctx.status = 200;
      ctx.body = { status: 'duplicate' };
      return;
    }

    // Queue for background processing (fast)
    await webhookQueue.add({
      webhookData,
      requestId,
      receivedAt: new Date().toISOString(),
    }, {
      attempts: 5,
      backoff: {
        type: 'exponential',
        delay: 2000, // 2s, 4s, 8s, 16s, 32s
      },
      removeOnComplete: 100, // Keep last 100 successful jobs
      removeOnFail: 1000, // Keep last 1000 failed jobs
    });

    // Return immediate 200 OK
    ctx.status = 200;
    ctx.body = {
      status: 'queued',
      requestId,
      timestamp: new Date().toISOString(),
    };
  } catch (error) {
    // Return 500 for recoverable errors (MP will retry)
    strapi.log.error('Webhook queueing error:', error);
    ctx.status = 500;
    ctx.body = { error: 'Internal server error' };
  }
}
```

**Alternative (Simpler)**:

Use Strapi's built-in job scheduling (no Redis required):

```typescript
// Use process.nextTick for immediate async processing
async handleWebhook(ctx: Context): Promise<void> {
  // ... signature validation ...

  // Process asynchronously (won't block HTTP response)
  process.nextTick(async () => {
    try {
      await this.handlePaymentNotification(webhookData.data, requestId);
    } catch (error) {
      strapi.log.error('Webhook processing error:', error);
      // Store in dead letter queue for manual retry
      await this.addToDeadLetterQueue(requestId, webhookData, error);
    }
  });

  // Return immediate 200 OK
  ctx.status = 200;
  ctx.body = { status: 'received', requestId };
}
```

```typescript
// Update webhook-handler.ts to use database
async addToDeadLetterQueue(
  requestId: string,
  webhookData: MPWebhookPayload,
  error: Error
): Promise<void> {
  try {
    const now = new Date();
    const nextRetry = new Date(now.getTime() + 5 * 60 * 1000); // Retry in 5 minutes

    await strapi.documents('api::webhook-failed-job.webhook-failed-job').create({
      data: {
        requestId,
        webhookType: webhookData.type,
        webhookData: webhookData as any,
        error: {
          message: error.message,
          stack: error.stack,
        },
        attemptCount: 1,
        maxAttempts: 5,
        nextRetryAt: nextRetry.toISOString(),
        firstAttempt: now.toISOString(),
        lastAttempt: now.toISOString(),
        status: 'pending',
      },
    });

    strapi.log.error('Webhook added to dead letter queue:', {
      requestId,
      nextRetryAt: nextRetry.toISOString(),
    });
  } catch (dbError) {
    strapi.log.error('Failed to store webhook in dead letter queue:', dbError);
    // Fall back to in-memory queue if database fails
    this.webhookQueue.push({
      requestId,
      webhookData,
      error: { message: error.message, stack: error.stack },
      attemptCount: 1,
      firstAttempt: now.toISOString(),
      lastAttempt: now.toISOString(),
    });
  }
}

          // Mark as retrying
          await strapi.documents('api::webhook-failed-job.webhook-failed-job').update({
            documentId: job.documentId,
            data: { status: 'retrying' },
          });

          // Retry webhook processing
          await strapi.webhookHandler.processWebhookAsync(
            job.webhookData,
            job.requestId,
            Date.now()
          );

          // Success - mark as resolved
          await strapi.documents('api::webhook-failed-job.webhook-failed-job').update({
            documentId: job.documentId,
            data: { status: 'resolved' },
          });

          strapi.log.info(`Successfully retried webhook ${job.requestId}`);
        } catch (error) {
          // Failed again - increment attempt count
          const attemptCount = job.attemptCount + 1;
          const exponentialBackoff = Math.pow(2, attemptCount) * 60 * 1000; // 2min, 4min, 8min, 16min, 32min
          const nextRetryAt = new Date(Date.now() + exponentialBackoff);

          await strapi.documents('api::webhook-failed-job.webhook-failed-job').update({
            documentId: job.documentId,
            data: {
              attemptCount,
              lastAttempt: new Date().toISOString(),
              nextRetryAt: nextRetryAt.toISOString(),
              status: attemptCount >= 5 ? 'failed' : 'pending',
              error: {
                ...job.error,
                lastError: error.message,
              },
            },
          });

          if (attemptCount >= 5) {
            strapi.log.error(`Webhook ${job.requestId} failed after ${attemptCount} attempts - manual intervention required`);
          }
        }
      }
    },
    options: {
      rule: '*/5 * * * *', // Every 5 minutes
    },
  },
});
```
